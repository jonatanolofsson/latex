\newcommand{\currentchapter}{}
\newcommand{\image}[2]{%
  \begin{figure}[H]%
    \centering%
    \includegraphics[width=#1\linewidth]{\currentchapter/#2}%
  \end{figure}%
}
\newcommand{\fig}[4]{%
  \begin{figure}[H]%
    \centering%
    \includegraphics[width=#1\linewidth]{\currentchapter/#2}%
    \caption{#3}\label{#4}%
  \end{figure}%
}
\newcommand{\tikzfig}[4]{%
  \begin{figure}[H]%
    \centering%
    \resizebox{#1\linewidth}{!}{\input{\currentchapter/#2}}%
    \caption{#3}\label{#4}%
  \end{figure}%
}

%% Math functions
\renewcommand{\sin}[1]{sin\left(#1\right)}
\renewcommand{\cos}[1]{cos\left(#1\right)}

%% Initialize new chapter
\newcommand{\newchapter}[1]{%
	\renewcommand{\currentchapter}{#1}
	\include*{#1/#1}
    \clearpage
}
